\newcommand*{\Rectangle}[4][]{%
    % #1 = draw/fill options
    % #2 = width
    % #3 = height
    % #4 = number of vertical lines

    \pgfmathsetmacro{\DeltaX}{#2/#4}%
    \foreach \x in {1,...,#4} {
        \pgfmathsetmacro{\NewX}{\DeltaX*\x}
        \draw ($(0,0)+(\NewX,0)$) -- ($(0,0)+(\NewX,#3)$);
    }
}
